46 template<typename T1, typename T2>
47 bool do_which_first(DERIVED_SPACE<T1>& s1, DERIVED_SPACE<T2>& s2) {
48  // We assume transformations are AOs on rows and MOs on columns and that we
49  // are transforming from AOs to MOs (N.B. there's nothing here tied to AOs
50  // or MOs)
51  auto s1_extent = s1.C().elements_range().extent();
52  const auto s1_nao = s1_extent[0];
53  const auto s1_nmo = s1_extent[1];
54 
55  auto s2_extent = s2.C().elements_range().extent();
56  const auto s2_nao = s2_extent[0];
57  const auto s2_nmo = s2_extent[1];
58 
59  //These transformations can only change the size of two of the modes of the
60  //tensor. When we start the tensor is s1_nao * s2_nao * x elements large
61  //(x is the product of the extents of the remaining modes). If we apply
62  // s1 first the result is s1_nmo * s2_nao * x and if we apply s2 first the
63  // result is s1_nao * s2_nmo * x. Assuming all quantities are integral and
64  // non-zero this means that if s1_nmo * s2_nao < s1_nao * s2_nmo we should
65  // apply s1 first. If s1_nmo * s2_nao > s1_nao * s2_nmo we should apply s2
66  // first. And if they're the same it doesn't matter which one gets applied
67  // first.
68  const auto s1_first = s1_nmo * s2_nao;
69  const auto s2_first = s1_nao * s2_nmo;
70  return s2_first < s1_first;
71 }

93 template<typename T1, typename T2, typename T3>
94 std::size_t do_which_first(DERIVED_SPACE<T1>& s1, DERIVED_SPACE<T2>& s2,
95  DERIVED_SPACE<T3>& s3) {
96  // This first part is conceptually all the same as the two input overload.
97  // See the comments there for more help.
98 
99  auto s1_extent = s1.C().elements_range().extent();
100  const auto s1_nao = s1_extent[0];
101  const auto s1_nmo = s1_extent[1];
102 
103  auto s2_extent = s2.C().elements_range().extent();
104  const auto s2_nao = s2_extent[0];
105  const auto s2_nmo = s2_extent[1];
106 
107  auto s3_extent = s3.C().elements_range().extent();
108  const auto s3_nao = s3_extent[0];
109  const auto s3_nmo = s3_extent[1];
110 
111  const auto s1_first = s1_nmo * s2_nao * s3_nao;
112  const auto s2_first = s1_nao * s2_nmo * s3_nao;
113  const auto s3_first = s1_nao * s2_nao * s3_nmo;
114 
115  // Theres's six possibilities:
116  // s1 <= s2 <= s3
117  // s1 <= s3 < s2 (s3 == s2 accounted for above)
118  // s2 < s1 <= s3 (s2 == s1 accounted for above)
119  // s2 <= s3 < s1 (s3 == s1 accounted for above)
120  // s3 < s1 <= s2 (s3 == s1 accounted for above)
121  // s3 < s2 < s1 (s3 == s1 and/or s2 == s1 accounted for above)
122  if(s1_first <= s2_first) {
123  // if ternary condition is true we have s3 < s1 <= s2
124  // if false, we have s1 <= s3 && s1 <= s2, which covers
125  // s1 <= s2 <= s3 && s1 <= s3 < s2
126  return s3_first < s1_first ? 2 : 0;
127  }
128  // We know s2 < s1. If ternary condition is true we have s2 < s1 &&
129  // s2 <= s3, which covers s2 < s1 <= s3 && s2 <= s3 < s1. If false, we have
130  // s3 < s2 < s1
131  return s2_first <= s3_first ? 1 : 2;
132 }

134 template<typename T1, typename T2, typename T3, typename T4>
135 std::size_t do_which_first(DERIVED_SPACE<T1>& s1, DERIVED_SPACE<T2>& s2,
136  DERIVED_SPACE<T3>& s3, DERIVED_SPACE<T4>& s4) {
137 
138  auto first = do_which_first(s1, s2, s3);
139  if(first == 0) {
140  // we know s1 <= s2 && s1 <= s3. If condition is false we also have
141  // s1 < s4. If true we have s4 <= s1 <= s2 && s4 <= s1 <= s3.
142  const bool s4_before_s1 = do_which_first(s1, s4);
143  return s4_before_s1? 3 : 0;
144  } else if(first == 1) {
145  // we know s2 < s1 && s2 <= s3. If condition is true we also have
146  // s2 <= s4. If false we have s4 < s2 < s1 && s4 < s2 <= s3.
147  const bool s2_before_s4 = do_which_first(s4, s2);
148  return s2_before_s4 ? 1 : 3;
149  }
150  // we know s3 < s1 and s3 < s2. If condition is true we also have
151  // s3 <= s4. If false we have s4 < s3 < s1 and s4 < s3 < s2.
152  const bool s3_before_s4 = do_which_first(s4, s3);
153  return s3_before_s4 ? 2 : 3;
154 }
