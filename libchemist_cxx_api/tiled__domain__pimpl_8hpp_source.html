216  // Make sure other is TiledDomainPIMPL
217  const auto* pother = downcast_(other);
218  if(pother == nullptr) {
219  throw std::runtime_error("RHS of operator*= is not a TiledDomainPIMPL");
220  }
221  // If both have TiledRanges take the product
222  if(has_trange_() && pother->has_trange_()) {
223  std::vector<TA::TiledRange1> tr1s;
224  for(std::size_t i = 0; i < this->rank(); ++i)
225  tr1s.push_back(m_trange_.dim(i));
226  for(std::size_t i = 0; i < other.rank(); ++i)
227  tr1s.push_back(pother->m_trange_.dim(i));
228  m_trange_ = TA::TiledRange(tr1s.begin(), tr1s.end());
229  } else if(other.size() == 0 && other.rank() == 0) { // Other is empty
230  m_trange_ = TA::TiledRange{};
231  } else if(this->size() == 0 && this->rank() == 0) { // I'm empty
232  return *this;
233  } else if(has_trange_()) {
234  throw std::runtime_error(
235  "LHS of operator*= has TiledRange, but RHS doesn't");
236  } else if(pother->has_trange_()) {
237  throw std::runtime_error(
238  "RHS of operator*= has TiledRange, but LHS doesn't");

241  BaseType::prod_assign_(other);
