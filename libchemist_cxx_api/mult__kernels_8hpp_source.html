43  static auto use_einsum_(Result& result, const LHS& lhs, const RHS& rhs) {
44  using utilities::strings::split_string;
45  const auto& lidx = split_string(lhs.annotation(), ",");
46  const auto& ridx = split_string(rhs.annotation(), ",");
47  const auto& oidx = split_string(result.annotation(), ",");
48 
49  for(const auto& x : oidx) {
50  const auto l_count = std::count(lidx.begin(), lidx.end(), x);
51  const auto r_count = std::count(ridx.begin(), ridx.end(), x);
52  if(l_count == 1 && r_count == 1) {
53  const auto& ostr = result.annotation();
54  const auto& lstr = lhs.annotation();
55  const auto& rstr = rhs.annotation();
56  auto& lt = lhs.array();
57  auto& rt = rhs.array();
58  auto& ot = result.array();
59  ot = ta_helpers::einsum::einsum(ostr, lstr, rstr, lt, rt);
60  return std::make_pair(true, result);
61  }
62  }
63  return std::make_pair(false, result);
64  }

98  template<typename ResultType, typename LHSType, typename RHSType>
99  static return_type eval(ResultType&& result, LHSType&& lhs, RHSType&& rhs) {
100  auto lhs_variant = lhs.variant(result);
101  auto rhs_variant = rhs.variant(result);
102  auto result_variant = result.variant(result);
103  using result_variant_t = std::decay_t<decltype(result_variant)>;
104  auto l = [&](auto&& result) {
105  auto m = [&](auto&& lhs) {
106  auto n = [&](auto&& rhs) {
107  auto [done, temp] = use_einsum_(result, lhs, rhs);
108  if(done) return result_variant_t(temp);
109  result = lhs * rhs;
110  return result_variant_t(result);
111  };
112  return std::visit(n, rhs_variant);
113  };
114  return std::visit(m, lhs_variant);
115  };
116  return std::visit(l, result_variant);
117  }
