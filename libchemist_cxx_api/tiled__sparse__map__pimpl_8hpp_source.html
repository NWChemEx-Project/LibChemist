190 typename TSMPIMPL::base_pimpl& TSMPIMPL::dp_assign_(const base_pimpl& rhs) {
191  const auto* pother = dynamic_cast<const my_type*>(&rhs);
192 
193  if(pother == nullptr) {
194  throw std::runtime_error(
195  "RHS of direct_product is not a TiledSparseMapPIMPL");
196  }
197  // If both have TiledRanges take the product
198  if(has_trange_() && pother->has_trange_()) {
199  std::vector<TA::TiledRange1> tr1s;
200  for(std::size_t i = 0; i < this->ind_rank(); ++i)
201  tr1s.push_back(m_trange_.dim(i));
202  for(std::size_t i = 0; i < rhs.ind_rank(); ++i)
203  tr1s.push_back(pother->m_trange_.dim(i));
204  m_trange_ = TA::TiledRange(tr1s.begin(), tr1s.end());
205  } else if(rhs.size() == 0 && rhs.ind_rank() == 0) { // Other is empty
206  m_trange_ = TA::TiledRange{};
207  } else if(this->size() == 0 && this->ind_rank() == 0) { // I'm empty
208  return *this;
209  } else if(has_trange_()) {
210  throw std::runtime_error(
211  "LHS of direct_product has TiledRange, but RHS doesn't");
212  } else if(pother->has_trange_()) {
213  throw std::runtime_error(
214  "RHS of direct_product has TiledRange, but LHS doesn't");
215  }
216 
217  return BaseType::dp_assign_(rhs);
218 }
