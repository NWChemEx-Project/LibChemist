214 template<typename TransformType, typename FromSpace, typename BaseType>
215 typename DERIVED_SPACE::overlap_type DERIVED_SPACE::compute_overlap_() const {
216  auto& S_ao = from_space().S();
217  auto& C_ao2x = C();
218  overlap_type S_deriv;
219 
220  using tile_type = typename FromSpace::overlap_type::value_type;
221  constexpr bool tile1_is_tot = TA::detail::is_tensor_of_tensor_v<tile_type>;
222 
223  using tile_type2 = typename std::decay_t<decltype(C_ao2x)>::value_type;
224  constexpr bool tile2_is_tot = TA::detail::is_tensor_of_tensor_v<tile_type2>;
225 
226  static_assert(
227  tile1_is_tot == tile2_is_tot,
228  "C and S are assumed to either both be tensors or both be ToTs.");
229 
230  if constexpr(tile1_is_tot) {
231  auto n_ind = S_ao.trange().rank() / 2;
232  std::string lidx, ridx;
233  for(std::size_t i = 0; i < n_ind; ++i) {
234  if(i > 0) {
235  lidx += ",";
236  ridx += ",";
237  }
238  lidx += "i_" + std::to_string(i);
239  ridx += "j_" + std::to_string(i);
240  }
241  std::string tot_idx = lidx + "," + ridx;
242 
243  auto ij_an = tot_idx + ";a,nu";
244  auto ij_ab = tot_idx + ";a,b";
245  auto ij_mn = tot_idx + ";mu,nu";
246  auto i_ma = lidx + ";mu,a";
247  auto j_nb = ridx + ";nu,b";
248 
249  overlap_type buffer;
250  TA::expressions::einsum(buffer(ij_an), C_ao2x(i_ma), S_ao(ij_mn));
251  TA::expressions::einsum(S_deriv(ij_ab), buffer(ij_an), C_ao2x(j_nb));
252  } else {
253  overlap_type buffer;
254  buffer("i,nu") = C_ao2x("mu,i") * S_ao("mu,nu");
255  S_deriv("i,j") = buffer("i,nu") * C_ao2x("nu,j");
256  }
257  return S_deriv;
258 }
