88 auto grab_diagonal(TensorType&& t) {
89  auto trange1 = t.trange().dim(0);
90  if(trange1 != t.trange().dim(1))
91  throw std::runtime_error("Expected a square tiling");
92  TA::TiledRange trange{{trange1}};
93  using tensor_type = std::decay_t<TensorType>;
94  using tile_type = typename tensor_type::value_type;
95  t.world().gop.fence();
96  auto l = [=](tile_type& tile, const TA::Range& r) {
97  auto idx = get_block_idx(trange, r);
98  auto t_tile = t.find({idx[0], idx[0]}).get();
99  tile = tile_type(r);
100  for(auto i : r) {
101  std::vector<std::size_t> ii{i[0], i[0]};
102  tile[i] = t_tile[ii];
103  }
104  return tile.norm();
105  };
106  auto rv = TA::make_array<tensor_type>(t.world(), trange, l);
107  t.world().gop.fence();
108  return rv;
109 }

279 template<typename T, typename U,
280  typename V = typename std::decay_t<T>::scalar_type>
281 bool allclose(T&& actual, U&& ref, const bool abs_comp = false,
282  V&& rtol = 1.0E-5, V&& atol = 1.0E-8,
283  std::size_t inner_rank = 0) {
284  using tensor_type = std::decay_t<T>;
285  using tile_type = typename tensor_type::value_type;
286  using scalar_type = std::decay_t<V>;
287  static_assert(std::is_same_v<tensor_type, std::decay_t<U>>,
288  "different tensor types is currently unsupported");
289 
290  constexpr bool is_tot = TA::detail::is_tensor_of_tensor_v<tile_type>;
291 
292  // Get a dummy string label (something like "i0, i1, i2, ...")
293  auto idx = TA::detail::dummy_annotation(actual.range().rank(), inner_rank);
294 
295  // Compute A - B, call result AmB
296  tensor_type AmB;
297  if(abs_comp) {
298  // TODO: There probably is some way to do this without forming the
299  // intermediate tensors
300  auto abs_lambda = [](const scalar_type& val) { return std::fabs(val); };
301  auto abs_actual = apply_elementwise(actual, abs_lambda);
302  auto abs_ref = apply_elementwise(ref, abs_lambda);
303  AmB(idx) = abs_actual(idx) - abs_ref(idx);
304  } else {
305  AmB(idx) = actual(idx) - ref(idx);
306  }
307 
308  auto times_op = [=](scalar_type lhs, scalar_type rhs) {
309  return std::fabs(lhs) <= atol + rtol * std::fabs(rhs);
310  };
311 
312  std::logical_and<bool> add_op;
313 
314  if constexpr(!is_tot) {
315  return reduce_elementwise(AmB, ref, add_op, times_op, true).get();
316  } else {
317  using inner_type = typename tile_type::value_type;
318  auto inner_times = [=](bool& result, const scalar_type& first,
319  const scalar_type& second) {
320  result = times_op(first, second);
321  };
322  auto outer_times = [=](inner_type lhs, inner_type rhs) {
323  return lhs.reduce(rhs, inner_times, add_op, true);
324  };
325  return reduce_elementwise(AmB, ref, add_op, outer_times, true,
326  inner_rank)
327  .get();
328  }
329 }
