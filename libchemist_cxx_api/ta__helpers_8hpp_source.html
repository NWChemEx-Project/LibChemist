216 template<typename T, typename U,
217  typename V = typename std::decay_t<T>::scalar_type>
218 bool allclose(T&& actual, U&& ref, V&& rtol = 1.0E-5, V&& atol = 1.0E-8,
219  std::size_t inner_rank = 0) {
220  using tensor_type = std::decay_t<T>;
221  using tile_type = typename tensor_type::value_type;
222  using scalar_type = std::decay_t<V>;
223  static_assert(std::is_same_v<tensor_type, std::decay_t<U>>,
224  "different tensor types is currently unsupported");
225 
226  constexpr bool is_tot = TA::detail::is_tensor_of_tensor_v<tile_type>;
227 
228  // Get a dummy string label (something like "i0, i1, i2, ...")
229  auto idx = TA::detail::dummy_annotation(actual.range().rank(), inner_rank);
230 
231  // Compute A - B, call result AmB
232  tensor_type AmB;
233  AmB(idx) = actual(idx) - ref(idx);
234 
235  auto times_op = [=](scalar_type lhs, scalar_type rhs) {
236  return std::fabs(lhs) <= atol + rtol * std::fabs(rhs);
237  };
238 
239  std::logical_and<bool> add_op;
240 
241  if constexpr (!is_tot) {
242  return reduce_elementwise(AmB, ref, add_op, times_op, true).get();
243  }
244  else{
245  using inner_type = typename tile_type::value_type;
246  auto inner_times = [=](bool& result, const scalar_type& first,
247  const scalar_type& second) {
248  result = times_op(first, second);
249  };
250  auto outer_times = [=](inner_type lhs, inner_type rhs) {
251  return lhs.reduce(rhs, inner_times, add_op, true);
252  };
253  return reduce_elementwise(AmB, ref, add_op, outer_times,
254  true, inner_rank).get();
255  }
256 }
