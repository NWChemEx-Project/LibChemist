447 SMPIMPL& SMPIMPL::dp_assign_(const my_type& rhs) {
448  if(m_sm_.empty() || rhs.m_sm_.empty()) {
449  m_sm_.clear();
450  return *this;
451  }
452  map_type new_map;
453 
454  using vector_type = std::vector<size_type>;
455  auto new_rank = ind_rank() + rhs.ind_rank();
456  for(auto [lkey, lval] : m_sm_){
457  for(const auto& [rkey, rval] : rhs.m_sm_) {
458  vector_type new_index;
459  new_index.reserve(new_rank);
460  new_index.insert(new_index.end(), lkey.begin(), lkey.end());
461  new_index.insert(new_index.end(), rkey.begin(), rkey.end());
462  auto new_domain = lval * rval;
463  if(new_domain.empty()) continue;
464  key_type key(new_index);
465  new_map.emplace(std::move(key), std::move(new_domain));
466  }
467  }
468  m_sm_.swap(new_map);
469  return *this;
470 }

473 SMPIMPL& SMPIMPL::prod_assign_(const my_type& rhs) {
474  if(m_sm_.empty()) return *this;
475  else if(rhs.m_sm_.empty()) {
476  m_sm_.clear();
477  return *this;

483  for(const auto& [lind, ldom] : m_sm_){
484  if(rhs.count(lind)){

486  if(new_dom.empty()) continue;
