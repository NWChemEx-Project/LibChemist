129 template<typename DerivedType>
130 bool ChargesViewPIMPL<ChargesType>::are_equal_impl_(
131  const ChargesViewPIMPL& other) const noexcept {
132  const auto* pthis = dynamic_cast<const DerivedType*>(this);
133  assert(pthis != nullptr); // Coding error if this happens
134  const auto* pother = dynamic_cast<const DerivedType*>(&other);
135 
136  // Same derived type can just delegate to operator==
137  if(pother != nullptr) { return (*pthis) == (*pother); }
138 
139  // Need to manually check through common API.
140 
141  // Short-circuit if different sizes
142  if(size() != other.size()) return false;
143 
144  // Compare elements in the range [0, size()), n.b. is no-op if size() == 0
145  for(size_type i = 0; i < size(); ++i)
146  if((*this)[i] != other[i]) return false;
147 
148  // Getting here means they're the same
149  return true;
