657 template<typename AtomicBasisSetType>
658 template<typename ABSType2, typename>
659 AtomicBasisSetView<AtomicBasisSetType>&
660 AtomicBasisSetView<AtomicBasisSetType>::operator=(const ABSType2& rhs) {
661  assert_non_null_();
662  if(rhs.is_null())
663  throw std::runtime_error(
664  "Assigning a null AtomicBasisSet to a non-null AtomicBasisSet is "
665  "currently not supported");
666  if(rhs.size() != this->size())
667  throw std::runtime_error(
668  "AtomicBasisSet to assign must be the same size as the aliased one");
669 
670  basis_set_name() = rhs.basis_set_name();
671  atomic_number() = rhs.atomic_number();
672  center().x() = rhs.center().x();
673  center().y() = rhs.center().y();
674  center().z() = rhs.center().z();
675  for(size_type shell = 0; shell < this->size(); ++shell) {
676  (*this)[shell] = rhs[shell];
677  }
678  return *this;
679 }
