5 // Transforming a tensor by a tensor
6 inline auto make_tensor_tensor_indices(const std::string& lhs,
7  const std::string& rhs,
8  std::size_t mode) {
9  using namespace utilities::strings;
10 
11  // The letter used in RHS's dummy indices
12  const std::string dummy_idx = "j";
13 
14  // Check mode is valid for rhs
15  const auto rhs_rank = split_string(rhs, ",").size();
16  if(rhs_rank <= mode) {
17  throw std::out_of_range("Can not transform zero-based mode " +
18  std::to_string(mode) + " of a tensor with ;" +
19  std::to_string(rhs_rank) + "modes.");
20  }
21 
22  // Check that lhs is a matrix
23  auto parsed_lhs = split_string(lhs, ",");
24  if(parsed_lhs.size() != 2)
25  throw std::runtime_error("Expected transformation to be matrix");
26 
27  // Assume column vectors
28  const std::string& new_idx = parsed_lhs[1];
29  const std::string& cont_idx = parsed_lhs[0];
30  std::string idx2replace = dummy_idx + std::to_string(mode);
31  auto rhs_idx = replace(idx2replace, cont_idx, rhs);
32  auto result_idx = replace(idx2replace, new_idx, rhs);
33  return std::make_pair(result_idx, rhs_idx);
34 }

51 inline auto make_tot_tot_indices(const std::string& lhs, const std::string& rhs,
52  std::size_t mode) {
53  // For C we can have:
54  // - "i;m" (or "i,j;m" if there's more independent indices), or
55  // - "i;m,a" (or "i,j;m,a" with more independent indices)
56  // We assume that C's independent indices are a subset of t's and that we're
57  // not contracting over the independent indices then:
58  // - "i;m" * "i,j;m,n,l,s" becomes "i,j;n,l,s"
59  // - "i,j;m" * "i,j;m,n,l,s" becomes "i,j;n,l,s"
60  // - "i;m,a" * "i,j;m,n,l,s" becomes "i,j;a,n,l,s"
61 
62  using namespace utilities::strings;
63 
64  // The letter used in RHS's dummy indices
65  const std::string rhs_dummy = "j";
66 
67  auto ltot_idx = split_string(lhs, ";");
68  auto lind_idx = split_string(ltot_idx[0], ",");
69  auto ldep_idx = split_string(ltot_idx[1], ",");
70  auto rtot_idx = split_string(rhs, ";");
71  auto rind_idx = split_string(rtot_idx[0], ",");
72  auto rdep_idx = split_string(rtot_idx[1], ",");
73 
74  if(mode < rind_idx.size())
75  throw std::runtime_error("ToT * ToT must transform dependent index.");
76 
77  if(ldep_idx.size() == 1)
78  throw std::runtime_error("C transforming to independent space NYI");
79 
80  if(lind_idx.size() != rind_idx.size())
81  throw std::runtime_error("Mixing independent ranks NYI");
82 
83  // The new rhs index is Hadamard-ed over lhs and rhs ind indices, and has
84  // mode "mode" replaced with ldep_idx[0]. The former is accomplished by
85  // making rhs's ind indices a copy of lhs's ind indices
86  std::string new_rhs = join_string(lind_idx, ",");
87  new_rhs += ";";
88  std::string result(new_rhs);
89  std::string idx2replace = rhs_dummy + std::to_string(mode);
90  new_rhs += replace(idx2replace, ldep_idx[0], rtot_idx[1]);
91  result += replace(idx2replace, ldep_idx[1], rtot_idx[1]);
92 
93  return std::make_pair(result, new_rhs);
94 }

124 inline auto make_indices(const std::string& lhs, const std::string& rhs,
125  std::size_t mode) {
126  // Determine if we're working with ToTs
127  const bool lhs_is_not_tot = lhs.find(";") == std::string::npos;
128  const bool lhs_is_tot = !lhs_is_not_tot;
129  const bool rhs_is_not_tot = rhs.find(";") == std::string::npos;
130  const bool rhs_is_tot = !rhs_is_not_tot;
131 
132  if(lhs_is_not_tot && rhs_is_not_tot)
133  return make_tensor_tensor_indices(lhs, rhs, mode);
134  else if(lhs_is_not_tot && rhs_is_tot)
135  return make_tensor_tot_indices(lhs, rhs, mode);
136  else if(lhs_is_tot && rhs_is_not_tot)
137  throw std::runtime_error("Can't transform a tensor by a ToT");
138 
139  return make_tot_tot_indices(lhs, rhs, mode);
140 }
