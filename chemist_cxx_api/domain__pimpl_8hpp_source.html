419 template<typename IndexType>

421  const value_type& old) const {
422  if(size() == 0 || old.size() != rank())
423  throw std::out_of_range("Index is not in domain");
424  std::vector<size_type> rv(rank(), 0);
425  for(std::size_t i = 0; i < rank(); ++i) {
426  const auto& offsets = m_mode_map_[i];
427  const auto value = old[i];
428  auto itr = std::find(offsets.begin(), offsets.end(), value);
429  if(itr == offsets.end())

431  rv[i] = std::distance(offsets.begin(), itr);
432  }

492  for(const auto& x : m_domain_) {

494  for(const auto& y : other.m_domain_) {

496  new_idx[i + rank()] = yi;
