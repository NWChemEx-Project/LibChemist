302 template<typename OpType>
303 LinearCombination& LinearCombination::add_term(OpType&& op) {
304  using clean_t = std::decay_t<OpType>;
305  static_assert(is_operator_v<clean_t>, "Argument must be an operator.");
306  const auto& rtti = typeid(clean_t);
307  add_term_(rtti, std::make_shared<clean_t>(std::forward<OpType>(op)));
308  return *this;
309 }

311 template<typename... OpTypes>
312 LinearCombination& LinearCombination::add_terms(OpTypes&&... ops) {
313  if constexpr(sizeof...(OpTypes) > 0) {
314  (add_term(std::forward<OpTypes>(ops)), ...);
315  }
316  return *this;
