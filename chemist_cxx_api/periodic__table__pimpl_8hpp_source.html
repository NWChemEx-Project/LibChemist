289 inline void PeriodicTablePIMPL::insert(size_type Z, const Atom& atom) {
290  if(m_atoms.count(Z))
291  throw std::runtime_error("Element already exists with Z = " +
292  std::to_string(Z));
293 
294  m_atoms.emplace(Z, std::move(atom));
295  m_isotopes.emplace(Z, atom_map());
296 
297  // Update max Z and symbol to Z mappings
298  if(Z > m_max_Z) m_max_Z = Z;
299  m_sym_2_Z.emplace(atom.name(), Z);
300 }

333  if(m_elec_confs.count(Z))

342  m_elec_confs.emplace(Z, std::move(clean_config));

368 inline typename PeriodicTablePIMPL::elec_conf_full_t
369 PeriodicTablePIMPL::reduced_2_full_conf(const elec_conf_t& r_conf) const {
370  elec_conf_full_t f_conf;
371  for(size_type l = 0; l < r_conf.size(); l++) {
372  size_type nmult = 2 * (2 * l + 1); // nelec in each shell
373  std::div_t d = std::div(r_conf[l], int(nmult));
374  size_type n_full_shells = d.quot;
375 
376  // fill starting from l+1 ([1s,2s,...], [2p, 3p, ...], [3d, 4d, ...],
377  // ...)
378  for(size_type i = 0; i < n_full_shells; i++) {
379  f_conf[{i + l + 1, l}] = nmult;
380  }
381  if(d.rem) f_conf[{n_full_shells + l + 1, l}] = d.rem;
382  }
383  return f_conf;
384 }

443 inline typename PeriodicTablePIMPL::elec_conf_t
444 PeriodicTablePIMPL::get_elec_conf_simple(size_type Z) const {
445  elec_conf_t conf(0);
446  size_t nrem = Z;
447  size_t nPlusl = 1;
448  while(nrem > 0) {
449  size_t lmax = (nPlusl - 1) / 2;
450  if(nPlusl % 2) { conf.push_back(0); }
451  for(size_t li = lmax; li <= lmax; li--) {
452  size_t delta_e = std::min(2 * (2 * li + 1), nrem);
453  conf[li] += delta_e;
454  nrem -= delta_e;
455  }
456  nPlusl += 1;
457  }
458  return conf;
459 }

461 inline typename PeriodicTablePIMPL::elec_conf_frac_t
462 PeriodicTablePIMPL::get_elec_conf_frac(double Z, double tol) const {
463  if(std::abs(Z - std::round(Z)) < tol) {
464  auto iconf = get_elec_conf(std::round(Z));
465  return elec_conf_frac_t(iconf.begin(), iconf.end());
466  }
467  size_t Z0 = static_cast<size_type>(Z);
468  size_t Z1 = Z0 + 1;
469  double dz = Z - Z0;
470 
471  auto iconf0 = get_elec_conf(Z0);
472  auto iconf1 = get_elec_conf(Z1);
473 
474  auto n0 = iconf0.size();
475  auto n1 = iconf1.size();
476 
477  elec_conf_frac_t conf(n1, 0);
478 
479  std::transform(iconf0.begin(), iconf0.end(), iconf1.begin(), conf.begin(),
480  [=](double i0, double i1) { return i0 + dz * (i1 - i0); });
481 
482  for(size_t ni = n0; ni < n1; ni++) { conf[ni] = iconf1[ni] * dz; }
483  return conf;
484 }
