162 template<typename ChargesType>
163 bool ChargesContiguous<ChargesType>::operator==(
164  const ChargesContiguous& rhs) const noexcept {
165  // Must have same size
166  if(size_() != rhs.size()) return false;
167 
168  // If they're both empty, then they are both the same
169  if(size_() == 0) return true;
170 
171  // Now we now they both have the same non-zero size
172 
173  // Start by comparing the points
174  if(m_points_ != rhs.m_points_) return false;
175 
176  // If aliasing the same memory for the charges then they're the same
177  if(m_pcharges_ == rhs.m_pcharges_) return true;
178 
179  // Have to manually compare the charges
180  return std::equal(m_pcharges_, m_pcharges_ + size_(), rhs.m_pcharges_);
181 }
