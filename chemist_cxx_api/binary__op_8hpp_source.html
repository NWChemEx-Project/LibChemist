207 template<typename DerivedType, typename LHSType, typename RHSType>
208 template<typename DerivedType2, typename LHSType2, typename RHSType2>
209 bool BinaryOp<DerivedType, LHSType, RHSType>::operator==(
210  const BinaryOp<DerivedType2, LHSType2, RHSType2>& other) const noexcept {
211  using lhs2_type = typename TermTraits<LHSType2>::value_type;
212  using rhs2_type = typename TermTraits<RHSType2>::value_type;
213  constexpr auto l_same = std::is_same_v<lhs_type, lhs2_type>;
214  constexpr auto r_same = std::is_same_v<rhs_type, rhs2_type>;
215  if constexpr(l_same && r_same) {
216  return std::tie(lhs(), rhs()) == std::tie(other.lhs(), other.rhs());
217  } else {
218  return false;
219  }
220 }
