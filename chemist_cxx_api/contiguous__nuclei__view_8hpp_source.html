182 template<typename NucleiType>
183 bool ContiguousNucleiView<NucleiType>::operator==(
184  const ContiguousNucleiView& rhs) const noexcept {
185  const auto n = size_();
186 
187  if(n != rhs.size_()) return false;
188  if(n == 0) return true;
189 
190  // n.b. we now know both have the same non-zero number of nuclei
191 
192  // For each pointer in *this we first check if they are the same address
193  // (which guarantees they are the same values). If they are not the same
194  // address we need to actually compare the values.
195  if(m_pnames_ != rhs.m_pnames_) {
196  const auto* plhs = m_pnames_;
197  const auto* prhs = rhs.m_pnames_;
198  if(!std::equal(plhs, plhs + n, prhs)) return false;
199  }
200 
201  if(m_patomic_numbers_ != rhs.m_patomic_numbers_) {
202  const auto* plhs = m_patomic_numbers_;
203  const auto* prhs = rhs.m_patomic_numbers_;
204  if(!std::equal(plhs, plhs + n, prhs)) return false;
205  }
206 
207  if(m_pmasses_ != rhs.m_pmasses_) {
208  const auto* plhs = m_pmasses_;
209  const auto* prhs = rhs.m_pmasses_;
210  if(!std::equal(plhs, plhs + n, prhs)) return false;
211  }
212 
213  return m_charges_ == rhs.m_charges_;
214 }
