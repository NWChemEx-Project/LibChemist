210 template<typename ShellType>
211 void ATOMIC_BASIS_SET_PIMPL::add_shell(
212  typename abs_traits::pure_type pure,
213  typename abs_traits::angular_momentum_type l,
214  typename abs_traits::const_cg_reference cg) {
215  // This shell will have index...
216  auto shell_i = size();
217 
218  // Add the purity/angular momentum
219  m_pure_.push_back(pure);
220  m_l_.push_back(l);
221 
222  // Record how many primitives the shell's cg has
223  m_prims_per_cg_.push_back(cg.size());
224 
226  m_cg_offsets_.push_back(m_coefs_.size());
227 
228  for(size_type i = 0; i < cg.size(); ++i) {
229  m_coefs_.push_back(cg.at(i).coefficient());
230  m_exps_.push_back(cg.at(i).exponent());
231  m_prim2shell_.push_back(shell_i);
232  }

235 template<typename ShellType>
236 bool ATOMIC_BASIS_SET_PIMPL::operator==(
237  const AtomicBasisSetPIMPL& rhs) const noexcept {
238  // N.B. cg_offsets and prim2shell exist for convenience. Their stati is
239  // fully determined by prims_per_cg
240  if(m_name_ != rhs.m_name_) return false;
241  if(m_z_ != rhs.m_z_) return false;
242  if(m_center_ != rhs.m_center_) return false;
243  if(m_prims_per_cg_ != rhs.m_prims_per_cg_) return false;
244  if(m_coefs_ != rhs.m_coefs_) return false;
245  if(m_exps_ != rhs.m_exps_) return false;
246  if(m_pure_ != rhs.m_pure_) return false;
247  if(m_l_ != rhs.m_l_) return false;
248  return true;
249 }
