203 template<typename T>
204 typename ShellPIMPL<T>::size_type ShellPIMPL<T>::size() const noexcept {
205  const auto L = m_l_.value();
206  const auto p_val = 2ul * L + 1ul;
207  // closed form for 2+L choose 2
208  const auto c_val = (L * L + 3ul * L + 2ul) / 2ul;
209  const bool is_pure = m_pure_.value() == ShellType::pure;
210  return is_pure ? p_val : c_val;
211 }
