447 template<typename LHSSetType, typename RHSSetType>
448 bool operator==(const Subset<LHSSetType>& lhs, const Subset<RHSSetType>& rhs) {
449  if constexpr(!std::is_same_v<decltype(lhs), decltype(rhs)>) {
450  return false;
451  } else {
452  if(lhs.object() != rhs.object()) return false;
453  if(lhs.size() != rhs.size()) return false;
454  using size_type = typename Subset<LHSSetType>::size_type;
455  for(size_type i = 0; i < lhs.size(); ++i)
456  if(lhs[i] != rhs[i]) return false;
457  return true;
458  }
459 }
