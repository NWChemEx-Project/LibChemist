119 bool PointSetViewPIMPL<PointSetType>::are_equal_impl_(
120  const PointSetViewPIMPL& other) const noexcept {
121  const auto* pthis = dynamic_cast<const DerivedType*>(this);
122  assert(pthis != nullptr); // Coding error if this happens
123  const auto* pother = dynamic_cast<const DerivedType*>(&other);
124 
125  // Same derived type can just delegate to operator==
126  if(pother != nullptr) { return (*pthis) == (*pother); }
127 
128  // Need to manually check through common API.
129 
130  // Short-circuit if different sizes
131  if(size() != other.size()) return false;
132 
133  // Compare elements in the range [0, size()), n.b. is no-op if size() == 0
134  for(size_type i = 0; i < size(); ++i)
135  if((*this)[i] != other[i]) return false;
136 
137  // Getting here means they're the same
138  return true;
