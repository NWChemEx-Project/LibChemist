397 void FragmentedNuclei<NucleiType>::insert(BeginItr begin, EndItr end) {
398  // If we put the arguments directly into a vector this is what would result
399  using vec_t = decltype(std::vector(begin, end));
400 
401  // These would be the elements
402  using input_type = typename vec_t::value_type;
403 
404  // Were we given indices or nuclei?
405  if constexpr(std::is_integral_v<input_type>) {
406  nucleus_index_set inputs(begin, end);
407  insert(std::move(inputs));
408  } else {
409  // Assumed to be Nucleus or NucleusView objects
410  using input_t = typename const_reference::reference_container;
411  input_t inputs(begin, end);
412  insert(const_reference(inputs));
413  }
