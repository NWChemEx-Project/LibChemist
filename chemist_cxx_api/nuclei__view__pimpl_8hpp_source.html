174 template<typename NucleiType>
175 template<typename DerivedType>
176 bool NucleiViewPIMPL<NucleiType>::are_equal_impl_(
177  const NucleiViewPIMPL& other) const noexcept {
178  const auto* pthis = dynamic_cast<const DerivedType*>(this);
179  assert(pthis != nullptr); // Coding error if this happens
180  const auto* pother = dynamic_cast<const DerivedType*>(&other);
181 
182  // Same derived type can just delegate to operator==
183  if(pother != nullptr) { return (*pthis) == (*pother); }
184 
185  // Need to manually check through common API.
186 
187  // Short-circuit if different sizes
188  if(size() != other.size()) return false;
189 
190  // Compare elements in the range [0, size()), n.b. is no-op if size() == 0
191  for(size_type i = 0; i < size(); ++i)
192  if(this->get_nuke(i) != other.get_nuke(i)) return false;
193 
194  // Getting here means they're the same
195  return true;
196 }
