406 template<typename SetType>
407 bool FAMILYOFSETS::disjoint() const noexcept {
408  auto e = end();
409  for(auto lhs = begin(); lhs != e; ++lhs) {
410  auto rhs = lhs;
411  ++rhs;
412  for(; rhs != e; ++rhs) {
413  if(!traits_type::disjoint(*lhs, *rhs)) return false;
414  }
415  }
416  return true;
417 }

419 template<typename SetType>
420 bool FAMILYOFSETS::operator==(const FamilyOfSets& rhs) const noexcept {
421  if(size() != rhs.size()) return false;
422  if(has_obj_() != rhs.has_obj_()) return false;
423  if(!has_obj_()) return true; // Both are empty without objects
424  if(object() != rhs.object()) return false;
425  for(size_type i = 0; i < size(); ++i)
426  if((*this)[i] != rhs[i]) return false;
427  return true;
428 }
