368 template<typename BraType, typename OperatorType, typename KetType>
369 template<typename BraType2, typename OperatorType2, typename KetType2>
370 bool BraKet<BraType, OperatorType, KetType>::operator==(
371  const BraKet<BraType2, OperatorType2, KetType2>& rhs) const noexcept {
372  constexpr bool same_bra_type = std::is_same_v<BraType, BraType2>;
373  constexpr bool same_op_type = std::is_same_v<OperatorType, OperatorType2>;
374  constexpr bool same_ket_type = std::is_same_v<KetType, KetType2>;
375 
376  if constexpr(same_bra_type && same_op_type && same_ket_type) {
377  return bra() == rhs.bra() && op().are_equal(rhs.op()) &&
378  ket() == rhs.ket();
379  } else {
380  return false;
381  }
382 }
