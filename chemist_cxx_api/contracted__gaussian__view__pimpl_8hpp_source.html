49  ContractedGaussianViewPIMPL(size_type n_prims,
50  typename cg_traits::coefficient_reference c0,
51  typename cg_traits::exponent_reference e0,
52  typename cg_traits::center_reference center) :
53  m_n_prims_(n_prims),
54  m_pcoefs_(&c0),
55  m_pexps_(&e0),
56  m_center_(std::move(center)) {}

58  // -------------------------------------------------------------------------

61 
62  // Number of primitives
63  auto size() const noexcept { return m_n_prims_; }
64 
65  // reference to mutable i-th primitive's coefficient
66  decltype(auto) coefficient(size_type i) { return m_pcoefs_[i]; }
67 
68  // reference to read-only i-th primitive coefficient
69  decltype(auto) coefficient(size_type i) const { return m_pcoefs_[i]; }
70 
71  // reference to mutable exponent for the i-th primitive
72  decltype(auto) exponent(size_type i) { return m_pexps_[i]; }
73 
74  // reference to read-only exponent for the i-th primitive
75  decltype(auto) exponent(size_type i) const { return m_pexps_[i]; }
76 
77  // mutable reference to where these primitives are centered
78  typename cg_traits::center_reference center() { return m_center_.value(); }
79 
80  // read-only reference to where these primitives are centered
81  typename cg_traits::const_center_reference center() const {
82  return m_center_.value();
83  }
84 
85  // mutable reference to i-th primitive
86  auto operator[](size_type i) {
87  return typename cg_traits::primitive_reference(coefficient(i),
88  exponent(i), center());
89  }
90 
91  // read-only reference to the i-th primitive
92  auto operator[](size_type i) const {
93  return typename cg_traits::const_primitive_reference(
94  coefficient(i), exponent(i), center());
95  }
96 
97  // -------------------------------------------------------------------------
98  // -- Utility
99  // -------------------------------------------------------------------------
100 
101  bool operator==(const ContractedGaussianViewPIMPL& rhs) const noexcept {
102  if(size() != rhs.size()) return false;
103  if(m_center_ != rhs.m_center_) return false;
104  for(auto i = 0; i < size(); ++i) {
105  if(coefficient(i) != rhs.coefficient(i)) return false;
106  if(exponent(i) != rhs.exponent(i)) return false;
107  }
108  return true;
109  }
110 
111 private:
113  size_type m_n_prims_ = 0;
114 
116  typename cg_traits::coefficient_pointer m_pcoefs_ = nullptr;
117 
119  typename cg_traits::exponent_pointer m_pexps_ = nullptr;
120 
122  std::optional<typename cg_traits::center_reference> m_center_;
123 };
