177 inline bool NucleiSubset<NucleiType>::operator==(
178  const NucleiSubset& rhs) const noexcept {
179  if(m_members_ != rhs.m_members_) return false;
180 
181  // Try to avoid comparing the Nuclei by checking for the same address
182  if(m_nuclei_ == rhs.m_nuclei_) return true;

184  // Above works if they're both null, now see if only one is null
185  if(is_null() || rhs.is_null()) return false;

187  // Both are non-null, so compare the Nuclei
